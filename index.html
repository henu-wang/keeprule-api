Quick Start

cURL 
# Get all investment principles
curl https://henu-wang.github.io/keeprule-api/api/principles.json
JavaScript 
// Fetch principles and get a random one
const res = await fetch('https://henu-wang.github.io/keeprule-api/api/principles.json');
const data = await res.json();

// Random principle
const random = data.principles[Math.floor(Math.random() * data.total)];
console.log(`"${random.text}" — ${random.author}`);

// Filter by author
const buffett = data.principles.filter(p => p.author === 'Warren Buffett');

// Filter by category
const risk = data.principles.filter(p => p.category === 'Risk Management');
Python 
import requests, random

data = requests.get('https://henu-wang.github.io/keeprule-api/api/principles.json').json()

# Random principle
p = random.choice(data['principles'])
print(f'"{p["text"]}" — {p["author"]}')

# Filter by author
munger = [p for p in data['principles'] if p['author'] == 'Charlie Munger']
